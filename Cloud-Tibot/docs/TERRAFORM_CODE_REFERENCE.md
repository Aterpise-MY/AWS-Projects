/**
 * Project CORTEX - EventBridge Rules Configuration
 * 
 * Creates EventBridge rule to monitor ALL AWS Amplify build status changes
 * (SUCCEED, FAILED, STARTED) across all Amplify apps in the account.
 * Triggers the Auto-Remediator Lambda function for:
 *   - Telegram notifications on all build status changes
 *   - AI-powered auto-remediation on build failures
 */

# -----------------------------------------------------------------------------
# EventBridge Rule for Amplify Build Status Changes (All Apps)
# -----------------------------------------------------------------------------

resource "aws_cloudwatch_event_rule" "amplify_build_status" {
  name        = "${var.project_name}_amplify_build_status"
  description = "Triggers Auto-Remediator Lambda on any Amplify build status change (all apps)"

  event_pattern = jsonencode({
    source      = ["aws.amplify"]
    detail-type = ["Amplify Deployment Status Change"]
    detail = {
      jobStatus = ["SUCCEED", "FAILED", "STARTED"]
    }
  })

  tags = {
    Name   = "${var.project_name}-amplify-build-status-rule"
    Module = "AutoRemediator"
  }
}

**Complete Code:**
```python
"""
Project CORTEX - Module 1: Amplify Build Notifier

Triggered by EventBridge on ANY AWS Amplify build status change
(STARTED, SUCCEED, FAILED) for ALL Amplify apps in the account.

Sends a Telegram notification with build details including:
  - App name and ID
  - Branch name and commit info
  - Build status with emoji indicators
  - Direct link to the Amplify console
"""

import json
import os
import boto3
import urllib3


def lambda_handler(event, context):
    """
    Main entry point. Receives EventBridge Amplify build status change events
    for ALL Amplify apps and sends a Telegram notification.
    """
    print(f"Received event: {json.dumps(event)}")

    detail = event.get("detail", {})
    app_id = detail.get("appId", "Unknown")
    branch_name = detail.get("branchName", "Unknown")
    job_id = detail.get("jobId", "Unknown")
    job_status = detail.get("jobStatus", "Unknown")
    commit_id = detail.get("commitId", "")

    telegram_token = os.environ["TELEGRAM_TOKEN"]
    telegram_chat_id = os.environ["TELEGRAM_CHAT_ID"]
    aws_region = os.environ.get("AWS_REGION", "us-east-1")

    # Map status to emoji
    status_emoji = {
        "SUCCEED": "✅",
        "FAILED": "❌",
        "STARTED": "🔄",
        "CANCELLING": "⏸️",
        "CANCELLED": "🚫",
    }.get(job_status, "ℹ️")

    # Get the app name
    app_name = app_id
    try:
        amplify_client = boto3.client("amplify")
        app_resp = amplify_client.get_app(appId=app_id)
        app_name = app_resp.get("app", {}).get("name", app_id)
    except Exception as e:
        print(f"Could not fetch app name: {e}")

    # Build console link
    console_url = (
        f"https://{aws_region}.console.aws.amazon.com/amplify/home"
        f"?region={aws_region}#/{app_id}/{branch_name}/{job_id}"
    )

    commit_display = f"`{commit_id[:7]}`" if commit_id else "N/A"

    # Compose message
    telegram_message = (
        f"{status_emoji} *Amplify Build Notification*\n"
        f"━━━━━━━━━━━━━━━━━━━━━━\n\n"
        f"*App:* `{app_name}`\n"
        f"*App ID:* `{app_id}`\n"
        f"*Branch:* `{branch_name}`\n"
        f"*Commit:* {commit_display}\n"
        f"*Job ID:* `{job_id}`\n"
        f"*Status:* *{job_status}* {status_emoji}\n\n"
        f"[View in AWS Console]({console_url})"
    )

    send_telegram_message(telegram_token, telegram_chat_id, telegram_message)

    return {
        "statusCode": 200,
        "body": json.dumps({
            "message": f"Telegram notification sent for {job_status}",
            "app_id": app_id,
            "app_name": app_name,
            "branch": branch_name,
            "status": job_status,
        }),
    }


def send_telegram_message(token, chat_id, message):
    """Send a message to Telegram Bot API."""
    http = urllib3.PoolManager()
    url = f"https://api.telegram.org/bot{token}/sendMessage"
    payload = {
        "chat_id": chat_id,
        "text": message[:4096],
        "parse_mode": "Markdown",
        "disable_web_page_preview": True,
    }
    try:
        response = http.request(
            "POST", url,
            body=json.dumps(payload).encode("utf-8"),
            headers={"Content-Type": "application/json"},
        )
        if response.status == 200:
            print("Telegram message sent successfully")
        else:
            print(f"Failed to send Telegram message: {response.data.decode('utf-8')}")
    except Exception as e:
        print(f"Error sending Telegram message: {str(e)}")
```
